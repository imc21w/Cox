struct Node{
    fun init(){
        println("node init");
    }

    fun look(a){
        if (isNull(a))
            a = "node";
        println(a, " ", "look");
    }
}

struct Child extends Node{
    fun init(){
        println("child init");
    }
}

struct Apple extends Child{
    fun init(){
        println("apple init");
    }

    fun set(){
        //super.a = 10;
    }

    fun look(){
        print super.a;
    }
}

let a = Apple();
println();
a.set();
a.look();
